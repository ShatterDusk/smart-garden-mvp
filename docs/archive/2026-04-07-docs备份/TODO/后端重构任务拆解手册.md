## 一、任务拆解原则

### 1.1 SMART原则
- **Specific**: 每个任务有明确的输入和输出
- **Measurable**: 验收标准可量化
- **Achievable**: 单个任务可在30分钟内完成
- **Relevant**: 任务间依赖关系清晰
- **Time-bound**: 每个任务有预估时间

### 1.2 任务粒度
- 单个任务代码量 < 100行
- 单个任务函数数 < 5个
- 单个任务涉及文件 < 3个

### Task 1.1: 创建 BaseService 基类

**输入**:
- 无

**输出**:
- 文件: `backend/server/src/services/BaseService.js`

**执行步骤**:
```
1. 创建文件 services/BaseService.js
2. 定义 BaseService 类
3. 实现 findById(id) 方法
4. 实现 findOne(conditions) 方法
5. 实现 findAll(conditions) 方法
6. 实现 create(data) 方法
7. 实现 update(id, data) 方法
8. 实现 delete(id) 方法
9. 实现 paginate(query) 方法（分页）
10. 导出 BaseService
```

**代码模板**:
```javascript
const logger = require('../utils/logger');

class BaseService {
  constructor(model, modelName) {
    this.model = model;
    this.modelName = modelName;
  }

  async findById(id) {
    try {
      return await this.model.findByPk(id);
    } catch (err) {
      logger.error(`${this.modelName}.findById error:`, err);
      throw err;
    }
  }

  async findOne(conditions) {
    try {
      return await this.model.findOne({ where: conditions });
    } catch (err) {
      logger.error(`${this.modelName}.findOne error:`, err);
      throw err;
    }
  }

  async findAll(conditions = {}, options = {}) {
    try {
      return await this.model.findAll({
        where: conditions,
        ...options,
      });
    } catch (err) {
      logger.error(`${this.modelName}.findAll error:`, err);
      throw err;
    }
  }

  async create(data) {
    try {
      return await this.model.create(data);
    } catch (err) {
      logger.error(`${this.modelName}.create error:`, err);
      throw err;
    }
  }

  async update(id, data) {
    try {
      const record = await this.findById(id);
      if (!record) return null;
      return await record.update(data);
    } catch (err) {
      logger.error(`${this.modelName}.update error:`, err);
      throw err;
    }
  }

  async delete(id) {
    try {
      const record = await this.findById(id);
      if (!record) return null;
      await record.destroy();
      return true;
    } catch (err) {
      logger.error(`${this.modelName}.delete error:`, err);
      throw err;
    }
  }

  async paginate(conditions = {}, page = 1, pageSize = 20) {
    try {
      const offset = (parseInt(page) - 1) * parseInt(pageSize);
      const limit = parseInt(pageSize);
      const { count, rows } = await this.model.findAndCountAll({
        where: conditions,
        offset,
        limit,
      });
      return {
        total: count,
        page: parseInt(page),
        pageSize: limit,
        list: rows,
      };
    } catch (err) {
      logger.error(`${this.modelName}.paginate error:`, err);
      throw err;
    }
  }
}

module.exports = BaseService;
```

### Task 4.1: 创建 DeviceService 类

**输入**:
- BaseService.js
- models/Device.js

**输出**:
- 文件: `backend/server/src/services/DeviceService.js`

**代码模板**:
```javascript
const BaseService = require('./BaseService');
const { Device, Plant } = require('../models');
const { v4: uuidv4 } = require('uuid');
const logger = require('../utils/logger');

class DeviceService extends BaseService {
  constructor() {
    super(Device, 'Device');
  }

  generateDeviceId() {
    return `DEVICE_${uuidv4().replace(/-/g, '').substring(0, 16)}`;
  }

  async bindDevice(userId, macAddress, plantId) {
    try {
      let device = await this.findOne({ mac_address: macAddress });
      if (!device) {
        device = await this.create({
          device_id: this.generateDeviceId(),
          mac_address: macAddress,
          device_name: `设备-${macAddress.slice(-4)}`,
          status: 'online',
          user_id: userId,
          bound_plant_id: plantId,
          created_at: new Date(),
        });
      } else {
        await device.update({
          user_id: userId,
          bound_plant_id: plantId,
          status: 'online',
        });
      }
      logger.info(`Device bound: ${macAddress} -> ${plantId}`);
      return device;
    } catch (err) {
      logger.error('DeviceService.bindDevice error:', err);
      throw err;
    }
  }

  async unbindDevice(deviceId) {
    try {
      const device = await this.findById(deviceId);
      if (!device) return null;
      await device.update({
        bound_plant_id: null,
        status: 'unbound',
      });
      logger.info(`Device unbound: ${deviceId}`);
      return device;
    } catch (err) {
      logger.error('DeviceService.unbindDevice error:', err);
      throw err;
    }
  }

  async getDeviceList(userId) {
    try {
      return await this.findAll({ user_id: userId });
    } catch (err) {
      logger.error('DeviceService.getDeviceList error:', err);
      throw err;
    }
  }

  async getDeviceById(deviceId) {
    try {
      return await this.findById(deviceId);
    } catch (err) {
      logger.error('DeviceService.getDeviceById error:', err);
      throw err;
    }
  }
}

module.exports = DeviceService;
```

      const recordedAtDate = new Date(recordedAt);

      let task = await ReadingTask.findOne({
        where: { plant_id: plantId, recorded_at: recordedAtDate },
      });

      if (!task) {
        task = await ReadingTask.create({
          task_id: `TASK_${uuidv4().replace(/-/g, '').substring(0, 16)}`,
          plant_id: plantId,
          recorded_at: recordedAtDate,
          sensor_status: 'pending',
          weather_status: 'failed',
        });
      }

      if (isSupplement && task.sensor_status === 'received') {
        return { error: '该时刻已有真实传感器数据，拒绝补传' };
      }

      if (task.sensor_status === 'compensated') {
        return await compensationService.coverCompensatedData(task, data);
      }

      const reading = await this.create({
        reading_id: this.generateReadingId(),
        plant_id: plantId,
        device_id: deviceId,
        recorded_at: recordedAtDate,
        data_source: 'sensor',
        is_stale: false,
        is_supplement: isSupplement || false,
        created_at: new Date(),
      });

      await task.update({ sensor_status: 'received' });
      logger.info(`EnvironmentReading created: ${reading.reading_id}`);
      return reading;
    } catch (err) {
      logger.error('EnvironmentService.uploadReading error:', err);
      throw err;
    }
  }

  async getCurrentData(plantId) {
    try {
      const reading = await this.model.findOne({
        where: { plant_id: plantId },
        order: [['recorded_at', 'DESC']],
      });
      return reading;
    } catch (err) {
      logger.error('EnvironmentService.getCurrentData error:', err);
      throw err;
    }
  }

  async getHistoryData(plantId, query = {}) {
    try {
      const { startTime, endTime, metricCode } = query;
      const conditions = { plant_id: plantId };
      if (startTime && endTime) {
        conditions.recorded_at = {
          [require('sequelize').Op.between]: [new Date(startTime), new Date(endTime)],
        };
      }
      return await this.findAll(conditions, { order: [['recorded_at', 'ASC']] });
    } catch (err) {
      logger.error('EnvironmentService.getHistoryData error:', err);
      throw err;
    }
  }
}

module.exports = EnvironmentService;

### Task 6.1: 创建 SessionService 类

**输入**:

- models/Session.js, models/Message.js

**输出**:

**代码模板**:
```javascript
const BaseService = require('./BaseService');
const { Session, Message } = require('../models');
const { v4: uuidv4 } = require('uuid');
const logger = require('../utils/logger');

class SessionService extends BaseService {
  constructor() {
    super(Session, 'Session');
  }

  generateSessionId() {
    return `SESSION_${uuidv4().replace(/-/g, '').substring(0, 16)}`;
  }

  generateMessageId() {
    return `MSG_${uuidv4().replace(/-/g, '').substring(0, 16)}`;
  }

  async createSession(userId, sessionData) {
    try {
      const sessionId = this.generateSessionId();
      const session = await this.create({
        session_id: sessionId,
        user_id: userId,
        session_type: sessionData.sessionType || 'consultation',
        plant_id: sessionData.plantId || null,
        title: sessionData.title || '新会话',
        status: 'active',
        created_at: new Date(),
        updated_at: new Date(),
      });
      logger.info(`Session created: ${sessionId}`);
      return session;
    } catch (err) {
      logger.error('SessionService.createSession error:', err);
      throw err;
    }
  }

  async getSessionList(userId, query = {}) {
    try {
      const { page = 1, pageSize = 20, sessionType } = query;
      const conditions = { user_id: userId };
      if (sessionType) conditions.session_type = sessionType;
      return await this.paginate(conditions, page, pageSize);
    } catch (err) {
      logger.error('SessionService.getSessionList error:', err);
      throw err;
    }
  }

  async getMessages(sessionId, query = {}) {
    try {
      const { page = 1, pageSize = 50 } = query;
      return await Message.findAndCountAll({
        where: { session_id: sessionId },
        order: [['created_at', 'ASC']],
        offset: (page - 1) * pageSize,
        limit: pageSize,
      });
    } catch (err) {
      logger.error('SessionService.getMessages error:', err);
      throw err;
    }
  }

  async sendMessage(sessionId, messageData) {
    try {
      const message = await Message.create({
        message_id: this.generateMessageId(),
        session_id: sessionId,
        role: messageData.role,
        content: messageData.content,
        content_type: messageData.contentType || 'text',
        image_urls: messageData.imageUrls || [],
        status: 'sent',
        created_at: new Date(),
      });
      await this.model.update({ updated_at: new Date() }, { where: { session_id: sessionId } });
      logger.info(`Message sent: ${message.message_id}`);
      return message;
    } catch (err) {
      logger.error('SessionService.sendMessage error:', err);
      throw err;
    }
  }

  async deleteSession(sessionId) {
    try {
      await Message.destroy({ where: { session_id: sessionId } });
      return await this.delete(sessionId);
    } catch (err) {
      logger.error('SessionService.deleteSession error:', err);
      throw err;
    }
  }

  async markAsRead(sessionId) {
    try {
      await Message.update({ status: 'read' }, { where: { session_id: sessionId } });
      return true;
    } catch (err) {
      logger.error('SessionService.markAsRead error:', err);
      throw err;
    }
  }
}

module.exports = SessionService;
```
