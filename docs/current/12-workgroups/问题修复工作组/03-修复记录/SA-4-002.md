const envConfig = {
  envId: process.env.WECHAT_ENV_ID,

## 修复方案

### 修复思路
1. 移除硬编码 bucket 名称
2. 强制从 `COS_BUCKET` 环境变量读取
3. 添加环境变量缺失校验，启动时抛出错误

### 修复后代码
```javascript
// 云托管环境配置
// 安全要求：COS_BUCKET 必须从环境变量读取，禁止硬编码
const envConfig = {
  envId: process.env.WECHAT_ENV_ID,
  bucket: process.env.COS_BUCKET,
  region: process.env.COS_REGION || 'ap-shanghai',
};

// 验证必需的环境变量
if (!envConfig.bucket) {
  throw new Error('COS_BUCKET 环境变量必须设置，用于云存储 bucket 名称');
}
```
