## 发现问题汇总

### 🔴 SA-8-001: 定时任务无并发控制（高危）

**文件位置**: `jobs/environmentSyncJob.js:L254-265`

**问题描述**:
```javascript
async function runSync() {
  logger.info('========== 环境数据同步开始 ==========');
  try {
    await generateTasksForAllPlants();  // 可能耗时较长
    await fetchWeatherForAllPlants();   // 可能耗时较长
    logger.info('========== 环境数据同步完成 ==========');
  } catch (error) {
    logger.error('环境数据同步失败', { error: error.message });
  }
}
```

**风险分析**:
- 如果 `runSync()` 执行时间超过2小时，下一次任务会在上一次未完成时启动
- 可能导致：
  - 数据库连接池耗尽
  - 重复生成相同的 reading_tasks
  - 天气 API 重复调用

**复现场景**:
1. 植物数量很多（>1000株）
2. 天气 API 响应慢
3. 网络波动导致请求重试

**修复建议**:
```javascript
let isRunning = false;

async function runSync() {
  if (isRunning) {
    logger.warn('上一次同步任务尚未完成，跳过本次执行');
    return;
  }
  
  isRunning = true;
  try {
    logger.info('========== 环境数据同步开始 ==========');
    await generateTasksForAllPlants();
    await fetchWeatherForAllPlants();
    logger.info('========== 环境数据同步完成 ==========');
  } catch (error) {
    logger.error('环境数据同步失败', { error: error.message });
  } finally {
    isRunning = false;
  }
}
```

**优先级**: 🔴 高
**状态**: ⏳ 待修复

### 🟠 SA-8-002: 无优雅关闭处理（中危）

**文件位置**: `app.js`

**问题描述**:
- 应用启动定时任务（`environmentSyncJob.start()`）
- 但没有注册 `process.on('SIGTERM')` 或 `process.on('SIGINT')` 处理
- 容器化部署时，收到停止信号可能导致正在执行的任务中断

**风险影响**:
- 数据不一致（任务执行到一半被中断）
- 数据库连接未正确关闭
- 日志不完整

**修复建议**:
```javascript
// app.js 末尾添加
const environmentSyncJob = require('./jobs/environmentSyncJob');

async function gracefulShutdown(signal) {
  logger.info(`收到 ${signal} 信号，开始优雅关闭...`);
  
  // 停止定时任务
  environmentSyncJob.stop();
  
  // 等待当前任务完成（设置超时）
  const shutdownTimeout = setTimeout(() => {
    logger.error('优雅关闭超时，强制退出');
    process.exit(1);
  }, 30000);
  
  // 关闭数据库连接
  await sequelize.close();
  logger.info('数据库连接已关闭');
  
  clearTimeout(shutdownTimeout);
  logger.info('优雅关闭完成');
  process.exit(0);
}

process.on('SIGTERM', () => gracefulShutdown('SIGTERM'));
process.on('SIGINT', () => gracefulShutdown('SIGINT'));
```

**优先级**: 🟠 中
**状态**: ⏳ 待修复

### 🟡 SA-8-003: 补偿任务无失败重试（低危）

**文件位置**: `services/compensationService.js:L78-101`

**问题描述**:
```javascript
async function checkAndCompensateAll() {
  try {
    const pendingTasks = await ReadingTask.findAll({...});
    for (const task of pendingTasks) {
      await compensateSensorReading(task);  // 失败即抛出，无重试
    }
  } catch (error) {
    logger.error('补偿检查失败', { error: error.message });
    throw error;
  }
}
```

**风险影响**:
- 单个任务失败会导致整个补偿流程中断
- 其他待补偿任务无法执行

**修复建议**:
```javascript
async function checkAndCompensateAll() {
  const pendingTasks = await ReadingTask.findAll({...});
  const results = { success: 0, failed: 0, errors: [] };
  
  for (const task of pendingTasks) {
    try {
      await compensateSensorReading(task);
      results.success++;
    } catch (error) {
      results.failed++;
      results.errors.push({ taskId: task.task_id, error: error.message });
      logger.error('单个任务补偿失败', { taskId: task.task_id, error: error.message });
      // 继续处理下一个，不中断
    }
  }
  
  logger.info('补偿检查完成', results);
  return results;
}
```

**优先级**: 🟡 低
**状态**: ⏳ 待优化
