## 1. 核心设计原则

### 1.1 为什么不动 AI 输入输出？

| 原因 | 说明 |
|:---|:---|
| **稳定性** | AI 提示词已经调优，改动可能降低诊断质量 |
| **复杂度** | 增加输出字段会增加 AI 解析负担 |
| **维护性** | 独立算法更容易测试和迭代 |
| **灵活性** | 可以基于多种数据源生成任务，不仅限于 AI 诊断 |

### 1.2 设计目标

- ✅ **零改动** AI 服务层代码
- ✅ **零改动** AI 提示词
- ✅ **零改动** AI 输出格式
- ✅ **纯后端算法** 处理诊断结果生成任务

### 3.1 诊断卡解析器（DiagnosisParser）

```javascript
// backend/server/src/algorithms/diagnosisParser.js

/**
 * 诊断卡解析器
 * 从 AI 返回的诊断卡中提取任务相关信息
 * 无需改动 AI 输出格式
 */
class DiagnosisParser {
  /**
   * 解析诊断卡
   * @param {Object} diagnosisCard - AI 返回的诊断卡
   * @returns {Object} 解析结果
   */
  parse(diagnosisCard) {
    if (!diagnosisCard) {
      return { hasIssues: false, tasks: [] };
    }

    const result = {
      hasIssues: false,
      healthScore: diagnosisCard.healthScore,
      status: diagnosisCard.status,
      confidence: diagnosisCard.confidence,
      tasks: []
    };

    // 1. 解析健康评分
    if (diagnosisCard.healthScore < 70) {
      result.hasIssues = true;
      result.tasks.push({
        type: 'health_alert',
        priority: diagnosisCard.healthScore < 50 ? 'high' : 'normal',
        reason: `健康评分 ${diagnosisCard.healthScore}，需要关注`
      });
    }

    // 2. 解析 issues
    if (diagnosisCard.issues && diagnosisCard.issues.length > 0) {
      result.hasIssues = true;
      diagnosisCard.issues.forEach(issue => {
        const task = this.parseIssue(issue);
        if (task) result.tasks.push(task);
      });
    }

    // 3. 解析 suggestions
    if (diagnosisCard.suggestions && diagnosisCard.suggestions.length > 0) {
      diagnosisCard.suggestions.forEach(suggestion => {
        const task = this.parseSuggestion(suggestion);
        if (task) result.tasks.push(task);
      });
    }

    return result;
  }

  /**
   * 解析 issue 为任务模板
   */
  parseIssue(issue) {
    const typeMap = {
      'watering': { taskType: 'water', title: '浇水' },
      'nutrition': { taskType: 'fertilize', title: '施肥' },
      'pruning': { taskType: 'prune', title: '修剪' },
      'repotting': { taskType: 'repot', title: '换盆' },
      'pest': { taskType: 'pestControl', title: '除虫' },
      'light': { taskType: 'adjustLight', title: '调整光照' },
      'temperature': { taskType: 'adjustTemp', title: '调整温度' },
      'humidity': { taskType: 'adjustHumidity', title: '调整湿度' }
    };

    const mapping = typeMap[issue.type];
    if (!mapping) return null;

    return {
      taskType: mapping.taskType,
      title: issue.name || mapping.title,
      description: issue.description,
      priority: this.mapSeverityToPriority(issue.severity),
      source: 'ai_issue',
      aiReason: issue.description
    };
  }

  /**
   * 解析 suggestion 为任务模板
   */
  parseSuggestion(suggestion) {
    const typeMap = {
      'watering': 'water',
      'nutrition': 'fertilize',
      'pruning': 'prune',
      'repotting': 'repot',
      'pestControl': 'pestControl',
      'environment': 'adjustEnv'
    };

    return {
      taskType: typeMap[suggestion.category] || 'other',
      title: suggestion.title,
      description: suggestion.content,
      priority: suggestion.priority || 'normal',
      source: 'ai_suggestion',
      aiReason: suggestion.reason || suggestion.content
    };
  }

  /**
   * 严重程度映射为优先级
   */
  mapSeverityToPriority(severity) {
    const map = {
      'high': 'high',
      'medium': 'normal',
      'low': 'low'
    };
    return map[severity] || 'normal';
  }
}

module.exports = DiagnosisParser;
```

### 3.2 任务生成器（TaskGenerator）

```javascript
// backend/server/src/algorithms/taskGenerator.js

const DiagnosisParser = require('./diagnosisParser');
const RuleEngine = require('./ruleEngine');

/**
 * 任务生成器
 * 整合诊断解析和规则引擎，生成最终任务
 */
class TaskGenerator {
  constructor() {
    this.parser = new DiagnosisParser();
    this.ruleEngine = new RuleEngine();
  }

  /**
   * 生成养护任务
   * @param {Object} params - 生成参数
   * @param {Object} params.diagnosisCard - AI 诊断卡
   * @param {Object} params.plantInfo - 植物信息
   * @param {Object} params.environmentData - 环境数据
   * @param {Array} params.existingTasks - 现有待办任务（用于去重）
   * @returns {Array} 生成的任务列表
   */
  async generate(params) {
    const { diagnosisCard, plantInfo, environmentData, existingTasks = [] } = params;
    
    const allTasks = [];

    // 1. 从诊断卡生成任务
    if (diagnosisCard) {
      const parsed = this.parser.parse(diagnosisCard);
      allTasks.push(...parsed.tasks.map(t => this.buildTask(t, plantInfo, 'ai')));
    }

    // 2. 从规则引擎生成任务
    const ruleTasks = await this.ruleEngine.generate(plantInfo, environmentData);
    allTasks.push(...ruleTasks.map(t => this.buildTask(t, plantInfo, 'rule')));

    // 3. 去重和过滤
    const filteredTasks = this.deduplicate(allTasks, existingTasks);

    // 4. 计算计划日期
    const finalTasks = filteredTasks.map(task => ({
      ...task,
      scheduledDate: this.calculateScheduledDate(task)
    }));

    return finalTasks;
  }

  /**
   * 构建任务对象
   */
  buildTask(template, plantInfo, source) {
    return {
      taskId: this.generateTaskId(),
      plantId: plantInfo.plantId,
      userId: plantInfo.userId,
      taskType: template.taskType,
      title: template.title,
      description: template.description,
      priority: template.priority,
      source: source,
      aiReason: template.aiReason,
      status: 'pending',
      createdAt: new Date().toISOString()
    };
  }

  /**
   * 任务去重
   * 避免同一天生成相同类型的多个任务
   */
  deduplicate(newTasks, existingTasks) {
    return newTasks.filter(newTask => {
      // 检查是否已存在相同类型的待办任务
      const hasDuplicate = existingTasks.some(existing => 
        existing.taskType === newTask.taskType &&
        existing.status === 'pending' &&
        this.isSameDay(existing.scheduledDate, newTask.scheduledDate)
      );
      
      return !hasDuplicate;
    });
  }

  /**
   * 计算计划执行日期
   */
  calculateScheduledDate(task) {
    const today = new Date();
    
    switch (task.priority) {
      case 'high':
        // 高优先级：今天
        return today;
      case 'normal':
        // 中优先级：明天
        return new Date(today.getTime() + 24 * 60 * 60 * 1000);
      case 'low':
        // 低优先级：3天内
        return new Date(today.getTime() + 3 * 24 * 60 * 60 * 1000);
      default:
        return today;
    }
  }

  /**
   * 生成任务ID
   */
  generateTaskId() {
    return `TASK_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }

  /**
   * 判断是否同一天
   */
  isSameDay(date1, date2) {
    const d1 = new Date(date1);
    const d2 = new Date(date2);
    return d1.toDateString() === d2.toDateString();
  }
}

module.exports = TaskGenerator;
```

   */
  getRules() {

      'succulent': {
        'spring': [
          { taskType: 'water', title: '浇水', frequency: 'weekly' }
        ],
        'summer': [
          { taskType: 'water', title: '控水', frequency: 'biweekly' }
        ],
        'autumn': [
          { taskType: 'water', title: '浇水', frequency: 'weekly' }
        ],
        'winter': [
          { taskType: 'water', title: '少量浇水', frequency: 'monthly' }
        ]
      },
      // 绿萝
      'pothos': {
        'spring': [
          { taskType: 'water', title: '浇水', frequency: 'weekly' },
          { taskType: 'fertilize', title: '施肥', frequency: 'monthly' }
        ],
        'summer': [
          { taskType: 'water', title: '浇水', frequency: 'biweekly' },
          { taskType: 'mist', title: '喷雾增湿', frequency: 'weekly' }
        ]
      }
      // ... 更多植物品种
    };
  }

  /**
   * 根据规则生成任务
   */
  async generate(plantInfo, environmentData) {
    const tasks = [];
    const season = this.getCurrentSeason();
    const rules = this.getRules();
    
    // 1. 品种规则
    const plantRules = rules[plantInfo.species]?.[season];
    if (plantRules) {
      plantRules.forEach(rule => {
        if (this.shouldGenerate(rule)) {
          tasks.push({
            taskType: rule.taskType,
            title: rule.title,
            priority: 'normal',
            source: 'rule',
            aiReason: `基于${plantInfo.species}的${season}养护规则`
          });
        }
      });
    }

    // 2. 环境数据规则
    if (environmentData) {
      const envTasks = this.generateFromEnvironment(environmentData);
      tasks.push(...envTasks);
    }

    return tasks;
  }

  /**
   * 从环境数据生成任务
   */
  generateFromEnvironment(envData) {
    const tasks = [];
    
    // 土壤湿度低
    if (envData.soilMoisture < 30) {
      tasks.push({
        taskType: 'water',
        title: '浇水',
        priority: 'high',
        source: 'rule',
        aiReason: `土壤湿度仅${envData.soilMoisture}%，建议浇水`
      });
    }
    
    // 光照不足
    if (envData.lightIntensity < 1000) {
      tasks.push({
        taskType: 'adjustLight',
        title: '调整光照',
        priority: 'normal',
        source: 'rule',
        aiReason: '光照强度不足，建议移至明亮处'
      });
    }
    
    return tasks;
  }

  /**
   * 获取当前季节
   */

    const month = new Date().getMonth() + 1;

    if (month >= 6 && month <= 8) return 'summer';

// backend/server/src/services/careTaskService.js

const TaskGenerator = require('../algorithms/taskGenerator');
const { CareTask } = require('../models');

/**
 * 养护任务服务
 */
class CareTaskService {
  constructor() {
    this.generator = new TaskGenerator();
  }

  /**
   * 从诊断卡生成任务
   * 在 AI 诊断完成后调用
   */
  async generateFromDiagnosis(diagnosisCard, plantInfo, environmentData) {
    // 1. 获取现有待办任务
    const existingTasks = await CareTask.findAll({
      where: {
        plantId: plantInfo.plantId,
        status: 'pending'
      }
    });

    // 2. 生成新任务
    const newTasks = await this.generator.generate({
      diagnosisCard,
      plantInfo,
      environmentData,
      existingTasks
    });

    // 3. 保存到数据库
    if (newTasks.length > 0) {
      await CareTask.bulkCreate(newTasks);
    }

    return newTasks;
  }

  /**
   * 获取今日待办
   */
  async getTodayTasks(userId) {
    const today = new Date().toISOString().split('T')[0];
    
    return await CareTask.findAll({
      where: {
        userId,
        scheduledDate: today,
        status: 'pending'
      },
      order: [
        ['priority', 'DESC'],
        ['createdAt', 'ASC']
      ],
      include: ['plant']
    });
  }

  /**
   * 完成任务
   */
  async completeTask(taskId, executionData) {
    const task = await CareTask.findByPk(taskId);
    if (!task) throw new Error('任务不存在');

    // 更新任务状态
    task.status = 'completed';
    task.completedAt = new Date();
    task.executionDescription = executionData.description;
    task.executionImages = executionData.images;
    
    await task.save();

    // 创建养护记录
    const careRecord = await this.createCareRecord(task, executionData);
    
    // 关联养护记录
    task.careRecordId = careRecord.recordId;
    await task.save();

    return task;
  }

  /**
   * 创建养护记录
   */
  async createCareRecord(task, executionData) {
    const { CareRecord } = require('../models');
    
    return await CareRecord.create({
      plantId: task.plantId,
      userId: task.userId,
      actionType: task.taskType,
      description: executionData.description,
      performedAt: new Date(),
      images: executionData.images,
      source: 'task',
      taskId: task.taskId
    });
  }
}

module.exports = new CareTaskService();
