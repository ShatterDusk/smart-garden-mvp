## 1. 任务生成方式概览

系统支持 **3 种** 任务生成方式：

| 方式 | 触发时机 | 优先级 | 复杂度 |
|:---|:---|:---:|:---:|
| **AI 智能生成** | 诊断后 / 定期分析 | 高 | 中 |
| **规则自动生成** | 新植物添加 / 周期性 | 中 | 低 |
| **用户手动创建** | 用户主动添加 | 低 | 低 |

---

## 2. AI 智能生成（主要方式）

### 2.1 触发时机

1. **用户完成 AI 诊断后** - 立即生成
2. **定期自动分析** - 每天凌晨分析所有植物
3. **环境数据异常** - 实时监测触发

### 2.2 生成逻辑

```javascript
/**
 * 基于诊断卡生成养护任务
 * @param {Object} diagnosisCard - AI 诊断卡
 * @param {String} plantId - 植物ID
 * @returns {Array} tasks - 生成的任务列表
 */
function generateTasksFromDiagnosis(diagnosisCard, plantId) {
  const tasks = [];
  
  // 1. 根据健康评分生成紧急任务
  if (diagnosisCard.healthScore < 60) {
    tasks.push({
      plantId,
      type: 'urgent_care',
      title: '紧急养护',
      priority: 'high',
      scheduledDate: getToday(),
      source: 'ai',
      aiReason: `健康评分仅${diagnosisCard.healthScore}分，需要立即关注`,
      status: 'pending'
    });
  }
  
  // 2. 根据 issues 生成具体任务
  diagnosisCard.issues.forEach(issue => {
    const task = mapIssueToTask(issue, plantId);
    if (task) tasks.push(task);
  });
  
  // 3. 根据 suggestions 生成长期任务
  diagnosisCard.suggestions.forEach(suggestion => {
    tasks.push({
      plantId,
      type: suggestion.category,
      title: suggestion.title,
      description: suggestion.content,
      priority: suggestion.priority,
      scheduledDate: calculateScheduledDate(suggestion),
      source: 'ai',
      aiReason: suggestion.reason || 'AI 智能建议',
      status: 'pending'
    });
  });
  
  return tasks;
}

/**
 * 将 issue 映射为任务
 */
function mapIssueToTask(issue, plantId) {
  const typeMap = {
    'watering': { type: 'water', title: '浇水' },
    'nutrition': { type: 'fertilize', title: '施肥' },
    'pruning': { type: 'prune', title: '修剪' },
    'repotting': { type: 'repot', title: '换盆' },
    'pest': { type: 'pestControl', title: '除虫' }
  };
  
  const mapping = typeMap[issue.type];
  if (!mapping) return null;
  
  return {
    plantId,
    type: mapping.type,
    title: issue.name || mapping.title,
    description: issue.description,
    priority: issue.severity === 'high' ? 'high' : 'normal',
    scheduledDate: getToday(),
    source: 'ai',
    aiReason: issue.description,
    status: 'pending'
  };
}
```

### 2.3 任务优先级映射

| 诊断卡字段 | 任务优先级 | 说明 |
|:---|:---:|:---|
| `healthScore < 60` | 🔴 high | 健康评分低，紧急 |
| `status === 'critical'` | 🔴 high | 严重状态 |
| `issue.severity === 'high'` | 🔴 high | 严重问题 |
| `issue.severity === 'medium'` | 🟡 normal | 中等问题 |
| `suggestion.priority === 'high'` | 🔴 high | 高优先级建议 |

## 3. 规则自动生成

### 3.1 触发时机

1. **新植物添加时** - 生成初始养护计划
2. **每周一凌晨** - 生成周期性任务
3. **季节变化时** - 调整养护频率

### 3.2 规则配置

```javascript

 * 养护规则配置
 * 基于植物品种和季节
 */
const CARE_RULES = {

  'succulent': {
    'spring': [
      { type: 'water', frequency: 'weekly', title: '浇水', dayOfWeek: 6 }
    ],
    'summer': [
      { type: 'water', frequency: 'biweekly', title: '控水', dayOfWeek: 0 }
    ],
    'autumn': [
      { type: 'water', frequency: 'weekly', title: '浇水', dayOfWeek: 6 }
    ],
    'winter': [
      { type: 'water', frequency: 'monthly', title: '少量浇水', dayOfWeek: 0 }
    ]
  },
  
  // 绿萝
  'pothos': {
    'spring': [
      { type: 'water', frequency: 'weekly', title: '浇水', dayOfWeek: 3 },
      { type: 'fertilize', frequency: 'monthly', title: '施肥', dayOfMonth: 1 }
    ],
    'summer': [
      { type: 'water', frequency: 'biweekly', title: '浇水', dayOfWeek: 3 },
      { type: 'mist', frequency: 'weekly', title: '喷雾增湿', dayOfWeek: 6 }
    ],
    'autumn': [
      { type: 'water', frequency: 'weekly', title: '浇水', dayOfWeek: 3 }
    ],
    'winter': [
      { type: 'water', frequency: 'biweekly', title: '控水', dayOfWeek: 3 }
    ]
  },
  
  // 月季
  'rose': {
    'spring': [
      { type: 'prune', frequency: 'once', title: '春季修剪', month: 3, day: 15 },
      { type: 'fertilize', frequency: 'biweekly', title: '施肥', dayOfWeek: 0 }
    ],
    'summer': [
      { type: 'water', frequency: 'daily', title: '浇水', checkTime: 'evening' },
      { type: 'pestControl', frequency: 'weekly', title: '病虫害检查', dayOfWeek: 0 }
    ]
  }
};

/**
 * 生成规则任务
 */
function generateRuleBasedTasks(plantSpecies, season, plantId) {
  const rules = CARE_RULES[plantSpecies]?.[season] || [];
  const tasks = [];
  
  rules.forEach(rule => {
    if (shouldGenerateTask(rule)) {
      tasks.push({
        plantId,
        type: rule.type,
        title: rule.title,
        priority: 'normal',
        scheduledDate: calculateScheduledDate(rule),
        source: 'rule',
        aiReason: `基于${plantSpecies}的${season}养护规则`,
        status: 'pending'
      });
    }
  });
  
  return tasks;
}
```

### 3.3 频率定义

| 频率 | 说明 | 示例 |
|:---|:---|:---|
| `daily` | 每天 | 夏季月季浇水 |
| `weekly` | 每周 | 多肉浇水 |
| `biweekly` | 每两周 | 冬季浇水 |
| `monthly` | 每月 | 施肥 |
| `once` | 一次性 | 春季修剪 |

## 4. 用户手动创建

### 4.1 使用场景

- 用户想自定义提醒
- 修改 AI 生成的任务时间
- 添加特殊养护需求

### 4.2 创建接口

```javascript
/**
 * 手动创建养护任务
 */
function createManualTask(params) {
  const {
    plantId,
    type,
    title,
    description,
    scheduledDate,
    priority = 'normal'
  } = params;
  
  return {
    taskId: generateUUID(),
    plantId,
    type,
    title,
    description,
    priority,
    scheduledDate,
    source: 'manual',
    status: 'pending',
    createdAt: new Date().toISOString()
  };
}
```

### 4.3 限制

- 仅手动创建的任务可以修改/删除
- AI 和规则生成的任务只能完成或跳过

## 5. 任务去重机制

### 5.1 去重规则

```javascript
/**
 * 任务去重
 * 避免同一天生成相同类型的多个任务
 */
function deduplicateTasks(newTasks, existingTasks) {
  return newTasks.filter(newTask => {
    // 检查是否已存在相同类型的待办任务
    const hasDuplicate = existingTasks.some(existing => 
      existing.type === newTask.type &&
      existing.status === 'pending' &&
      isSameDay(existing.scheduledDate, newTask.scheduledDate)
    );
    
    return !hasDuplicate;
  });
}
```

### 5.2 更新机制

如果 AI 生成了新的建议，但同类型任务已存在：
- **优先级更高**: 更新现有任务内容
- **优先级相同**: 保留现有任务
- **优先级更低**: 忽略新任务
