| 检查项 | 文档位置 | 代码位置 | 检查方法 |
|:---|:---|:---|:---|

| 必填字段一致 | 数据库设计.md | models/*.js | 对比 allowNull |
| 默认值一致 | 数据库设计.md | models/*.js | 对比 defaultValue |
| 关联关系一致 | 数据库设计.md | models/associations.config.js | 对比关联 |

### 2.2 检查脚本示例

```javascript
// 检查模型字段一致性
async function checkModelConsistency() {
  const issues = [];
  
  // 读取数据库设计文档
  const dbDoc = parseDatabaseDesignDoc();
  
  // 遍历所有模型
  for (const model of models) {
    const modelDef = require(`../models/${model}`);
    const docDef = dbDoc.tables[model];
    
    // 检查字段一致性
    for (const [fieldName, fieldDef] of Object.entries(modelDef.rawAttributes)) {
      if (!docDef.fields[fieldName]) {
        issues.push({
          type: 'MISSING_IN_DOC',
          model: model,
          field: fieldName,
          message: `字段 ${fieldName} 在文档中不存在`
        });
      }
    }
    
    // 检查文档中有但代码中无的字段
    for (const fieldName of Object.keys(docDef.fields)) {
      if (!modelDef.rawAttributes[fieldName]) {
        issues.push({
          type: 'MISSING_IN_CODE',
          model: model,
          field: fieldName,
          message: `字段 ${fieldName} 在代码中不存在`
        });
      }
    }
  }
  
  return issues;
}
```

### 7.2 枚举检查示例

```javascript
// 检查枚举一致性
function checkEnumConsistency() {
  const issues = [];
  
  const enumChecks = [
    {
      name: '植物分类',
      docValues: ['succulent', 'flower', 'foliage', 'vegetable', 'other'],
      codePath: 'backend/server/src/models/Plant.js',
      extractEnum: (content) => extractEnumFromModel(content, 'category')
    },
    // ... 其他枚举
  ];
  
  for (const check of enumChecks) {
    const codeContent = fs.readFileSync(check.codePath, 'utf8');
    const codeValues = check.extractEnum(codeContent);
    
    const diff = findDiff(check.docValues, codeValues);
    if (diff.length > 0) {
      issues.push({
        type: 'ENUM_MISMATCH',
        enum: check.name,
        differences: diff
      });
    }
  }
  
  return issues;
}
```

## 十、手动检查指南

### 10.1 何时需要手动检查

- 复杂业务流程的逻辑验证
- 涉及多个系统的交互验证
- 自动化工具无法覆盖的场景

### 10.2 手动检查步骤

```markdown
## 手动检查流程

1. **阅读文档**
   - 理解文档描述的功能/流程
   - 记录关键点和疑问

2. **定位代码**
   - 根据文档找到对应代码
   - 确认文件路径和函数名

3. **逐行对比**
   - 对比文档描述与代码实现
   - 标记不一致之处

4. **验证运行**
   - 如有条件，实际运行验证
   - 检查输出是否符合文档

5. **记录问题**
   - 使用标准格式记录
   - 提交到文档债跟踪清单
```
